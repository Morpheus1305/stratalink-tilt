export function generateSyntheticYesterday(today) {
  // Helper to apply controlled deltas (-8% → +8%)
  const jitter = (value, pct = 0.08) => {
    const delta = (Math.random() * 2 - 1) * pct; // -pct to +pct
    return {
      value_yesterday: value * (1 + delta),
      delta: delta * 100,
    };
  };

  return {
    depth10: jitter(today.depth10),
    depth25: jitter(today.depth25),
    depth50: jitter(today.depth50),
    factorScore: jitter(today.factorScore, 0.05), // ±5%
    stability: jitter(today.stability, 0.05),
    fragmentation: jitter(today.fragmentation, 0.10), // fragmentation is volatile
    executionRisk: jitter(today.executionRisk, 0.03),
    venues: jitter(today.venues, 0.10),
  };
}

export function generateCommentary(today, yesterday) {
  const depthTrend =
    yesterday.depth10.delta > 0
      ? "Depth improved modestly across tight bands."
      : "Depth softened, driven by reduced bid support at 10–25bps.";

  const factorTrend =
    yesterday.factorScore.delta > 0
      ? "Composite liquidity quality strengthened."
      : "Factor score deteriorated, led by stability and fragmentation pressure.";

  return `
  Institutional execution & liquidity posture for ${today.symbol}

  ${depthTrend}
  ${factorTrend}

  10bps depth: ${fmt(today.depth10)} (${deltaFmt(yesterday.depth10.delta)}).
  25bps depth: ${fmt(today.depth25)} (${deltaFmt(yesterday.depth25.delta)}).
  50bps depth: ${fmt(today.depth50)} (${deltaFmt(yesterday.depth50.delta)}).

  5-Factor Score: ${today.factorScore}/100 (${deltaFmt(yesterday.factorScore.delta)}).
  Fragmentation: ${today.fragmentation} (${deltaFmt(yesterday.fragmentation.delta)}).
  Venues: ${today.venues} (${deltaFmt(yesterday.venues.delta)}).
  `;
}

function deltaFmt(d) {
  return d >= 0 ? `▲ +${d.toFixed(2)}%` : `▼ ${d.toFixed(2)}%`;
}

function fmt(v) {
  return `$${Number(v).toLocaleString()}`;
}
