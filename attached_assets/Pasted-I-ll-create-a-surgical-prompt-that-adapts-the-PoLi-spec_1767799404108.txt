### 2. API Integration

**File:** `src/client/api.tsx` (or equivalent API service file)

Add new endpoint:
```typescript
export async function fetchPoLiData(
  token: string,
  window: '24H' | '7D' | '30D' = '7D',
  venues: string[] | 'ALL' = 'ALL'
): Promise<PoLiPayload> {
  const venueParam = venues === 'ALL' ? 'ALL' : venues.join(',');
  const response = await fetch(
    `/tilt/poli?token=${token}&window=${window}&venues=${venueParam}`
  );
  if (!response.ok) throw new Error('Failed to fetch PoLi data');
  return response.json();
}
```

export type RagStatus = 'GREEN' | 'AMBER' | 'ORANGE' | 'RED';
export type Confidence = 'HIGH' | 'MEDIUM' | 'LOW';
export type VerifyState = 'VALID' | 'WARNING' | 'INVALID' | 'PASS' | 'FAIL';

export interface PoLiPayload {
  meta: {
    token: string;
    window: string;
    venues: string[];
    last_updated: string;
    versions: {
      verify: string;
      poli: string;
      tsle: string;
    };
  };
  overall: {
    score: number;
    rag: RagStatus;
    confidence: Confidence;
    coverage: {
      venues_covered: number;
      venues_expected: number;
      regimes_observed: string[];
      min_required_regimes_met: boolean;
    };
    trend: {
      window_points: Array<{ t: string; score: number }>;
      delta_7d: number;
      direction: 'UP' | 'DOWN' | 'FLAT';
    };
  };
  pillars: {
    executability_cefi: PillarData;
    market_coherence: PillarData;
    onchain_executability: PillarData;
    method_integrity: PillarData;
  };
}

export interface PillarData {
  score: number;
  rag: RagStatus;
  summary: string;
  inputs: Record<string, InputCategory>;
}

export interface InputCategory {
  score: number;
  state: VerifyState;
  highlights?: string[];
  metrics?: Array<{
    key: string;
    label: string;
    value: number | string;
    unit: string;
  }>;
}

import React, { useState, useEffect } from 'react';
import { fetchPoLiData } from '../../client/api';
import { PoLiPayload } from '../../types/poli';
import PoLiScorecard from './PoLiScorecard';
import PillarSummaryBars from './PillarSummaryBars';
import PillarPanel from './PillarPanel';

export default function PoLiScorecardPage() {
  const [token, setToken] = useState('BTC');
  const [window, setWindow] = useState<'24H' | '7D' | '30D'>('7D');
  const [venues, setVenues] = useState<string[] | 'ALL'>('ALL');
  const [data, setData] = useState<PoLiPayload | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    setLoading(true);
    fetchPoLiData(token, window, venues)
      .then(setData)
      .catch(err => setError(err.message))
      .finally(() => setLoading(false));
  }, [token, window, venues]);

  if (loading) return <div>Loading PoLi data...</div>;
  if (error) return <div>Error: {error}</div>;
  if (!data) return null;

  return (
    <div className="poli-scorecard-page">
      {/* Header Controls */}
      <div className="header-controls">
        <select value={token} onChange={e => setToken(e.target.value)}>
          <option value="BTC">BTC</option>
          <option value="ETH">ETH</option>
          {/* Add more tokens */}
        </select>
        <select value={window} onChange={e => setWindow(e.target.value as any)}>
          <option value="24H">24H</option>
          <option value="7D">7D</option>
          <option value="30D">30D</option>
        </select>
      </div>

      {/* Row 1: Scorecard + Summary */}
      <div className="row-1">
        <PoLiScorecard overall={data.overall} meta={data.meta} />
        <PillarSummaryBars pillars={data.pillars} />
      </div>

      {/* Row 2: Pillar Panels */}
      <div className="pillar-grid">
        <PillarPanel
          title="Executability (CeFi)"
          pillarKey="executability_cefi"
          data={data.pillars.executability_cefi}
        />
        <PillarPanel
          title="Market Coherence"
          pillarKey="market_coherence"
          data={data.pillars.market_coherence}
        />
        <PillarPanel
          title="On-chain Executability"
          pillarKey="onchain_executability"

        <PillarPanel
          title="Method Integrity"

import React, { useState } from 'react';
import Card from './Card'; // REUSE
import { PillarData } from '../../types/poli';
import MetricRow from './MetricRow';

interface Props {
  title: string;
  pillarKey: string;
  data: PillarData;
}

export default function PillarPanel({ title, pillarKey, data }: Props) {
  const [expanded, setExpanded] = useState(false);

  return (
    <Card className="pillar-panel">
      <div className="pillar-header" onClick={() => setExpanded(!expanded)}>
        <h3>{title}</h3>
        <div className="pillar-score" data-rag={data.rag}>
          {data.score.toFixed(1)}
        </div>
      </div>

      <div className="pillar-summary">{data.summary}</div>

      {expanded && (
        <div className="pillar-inputs">
          {Object.entries(data.inputs).map(([inputKey, inputData]) => (
            <div key={inputKey} className="input-subcard">
              <div className="input-header">
                <span className="verify-tag">{inputKey.toUpperCase()}</span>
                <span className="input-score">{inputData.score.toFixed(1)}</span>
                <span className="input-state" data-state={inputData.state}>
                  {inputData.state}
                </span>
              </div>

              {inputData.highlights && (
                <ul className="highlights">
                  {inputData.highlights.map((h, i) => (
                    <li key={i}>{h}</li>
                  ))}
                </ul>
              )}

              {inputData.metrics && (
                <div className="metrics">
                  {inputData.metrics.map(m => (
                    <MetricRow key={m.key} metric={m} />
                  ))}
                </div>
              )}
            </div>
          ))}
        </div>
      )}
    </Card>
  );
}

### 7. Metric Row Component

**File:** `src/adapters/components/MetricRow.tsx` (new file)

```typescript
import React from 'react';

interface Metric {
  key: string;
  label: string;
  value: number | string;
  unit: string;
}

export default function MetricRow({ metric }: { metric: Metric }) {
  const formattedValue = typeof metric.value === 'number'
    ? metric.value.toFixed(metric.unit === '%' || metric.unit === 'bps' ? 1 : 2)
    : metric.value;

  return (
    <div className="metric-row">
      <span className="metric-label">{metric.label}</span>
      <span className="metric-value">
        {formattedValue}
        <span className="metric-unit">{metric.unit}</span>
      </span>
    </div>
  );
}
```
