/**
 * Format USD notional depth.
 * total_notional from LISBand is raw USD — adjust the divisor below if your
 * ingestion layer stores it in thousands or millions instead.
 */
function fmtDepth(v) {
  if (v == null || v === 0) return '—';
  if (v >= 1e9) return '$' + (v / 1e9).toFixed(1) + 'b';
  if (v >= 1e6) return '$' + (v / 1e6).toFixed(0) + 'm';
  if (v >= 1e3) return '$' + (v / 1e3).toFixed(0) + 'k';
  return '$' + Math.round(v);
}

function scoreColor(v) {
  return v >= 75 ? 'var(--green)' : v >= 55 ? 'var(--accent)' : 'var(--amber)';
}

function trendChar(prev, curr) {
  if (prev == null) return { sym: '→', cls: 'flat' };
  const d = curr - prev;
  if (d >  0.5) return { sym: '↑', cls: 'up'  };
  if (d < -0.5) return { sym: '↓', cls: 'dn'  };
  return { sym: '→', cls: 'flat' };
}

/**
 * Update a stress/regime tile.
 * Provide value=null for Phase 2 metrics not yet available.
 * gThresh / aThresh: value BELOW g = green, BELOW a = amber, else red.
 */
function applyTile(tile, value, formatter, gThresh, aThresh) {
  const valEl = tile.querySelector('.rt-value');
  const dotEl = tile.querySelector('.rt-dot');
  if (value == null) {
    // Already set in HTML — leave Phase 2 tiles alone
    return;

  const cls = value < gThresh ? 'g' : value < aThresh ? 'a' : 'r';
  tile.className = 'regime-tile ' + cls;
  valEl.textContent = formatter(value);
  valEl.style.color = '';
  dotEl.className = 'rt-dot ' + cls;
}

// ── Fetch & poll ──────────────────────────────────────────────────────────────

async function poll() {
  const t0 = performance.now();
  try {
    const res = await fetch(`/api/analytics/l5f/snapshot/${currentAsset}`);
    const latencyMs = Math.round(performance.now() - t0);
    document.getElementById('sb-lat').textContent = latencyMs + 'ms';

    if (!res.ok) {
      // 503 = buffer still warming; don't overwrite existing render
      return;
    }

    const json = await res.json();
    if (json.aggregate) {
      liveData[currentAsset] = json.aggregate;
      renderAggregate(json.aggregate);

      // Increment record counter as a proxy for throughput
      totalRecords += (json.aggregate.venue_count || 0) * 14;
      document.getElementById('sb-recs').textContent = totalRecords.toLocaleString();
    }
  } catch (e) {
    console.warn('[Terminal] poll error:', e);
  }
}

// ── Asset switch ──────────────────────────────────────────────────────────────

async function setAsset(a) {
  currentAsset = a;
  document.querySelectorAll('.asset-tab').forEach(t =>
    t.classList.toggle('active', t.textContent === a)
  );
  // Render from cache immediately for instant feedback, then refresh
  if (liveData[a]) renderAggregate(liveData[a]);
  await poll();
}

// ── Timestamp ticker ──────────────────────────────────────────────────────────

function updateTime() {
  const now = new Date();
  document.getElementById('tb-time').textContent =
    String(now.getHours()).padStart(2,'0')        + ':' +
    String(now.getMinutes()).padStart(2,'0')       + ':' +
    String(now.getSeconds()).padStart(2,'0')       + '.' +
    String(now.getMilliseconds()).padStart(3,'0');
}

poll();
