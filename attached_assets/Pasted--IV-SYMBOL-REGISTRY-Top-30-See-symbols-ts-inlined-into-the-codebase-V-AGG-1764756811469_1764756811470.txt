import axios from 'axios';

export async function fetchBinancePrice(symbol: string) {
  const url = \`https://api.binance.com/api/v3/ticker/price?symbol=\${symbol}\`;
  const { data } = await axios.get(url, { timeout: 700 });
  return Number(data.price);
}

import axios from 'axios';

export async function fetchKrakenPrice(pair: string) {
  const url = \`https://api.kraken.com/0/public/Ticker?pair=\${pair}\`;
  const { data } = await axios.get(url, { timeout: 1200 });
  const key = Object.keys(data.result)[0];
  return Number(data.result[key].c[0]);
}

# 5. FULL getAggregatedPrice() IMPLEMENTATION

cat > analytics/aggregator/aggregator.ts << 'EOF'
import { SYMBOL_MAP } from "./config/symbols";
import { fetchBinancePrice } from "./exchanges/binance";
import { fetchCoinbasePrice } from "./exchanges/coinbase";
import { fetchKrakenPrice } from "./exchanges/kraken";

const CACHE = new Map();
const TTL_MS = 3000;

export async function getAggregatedPrice({ symbol, quote = "USD" }) {
  const key = \`\${symbol}-\${quote}\`;

  // Cache
  const cached = CACHE.get(key);
  if (cached && Date.now() - cached.ts < TTL_MS)
    return { ...cached, cached: true };

  const map = SYMBOL_MAP[symbol];
  if (!map) throw new Error(\`Unsupported symbol: \${symbol}\`);

  const calls = [
    map.binance  && fetchBinancePrice(map.binance).then(p=>({name:"binance", price:p})).catch(()=>null),
    map.coinbase && fetchCoinbasePrice(map.coinbase).then(p=>({name:"coinbase",price:p})).catch(()=>null),
    map.kraken   && fetchKrakenPrice(map.kraken).then(p=>({name:"kraken", price:p})).catch(()=>null),
  ];

  const results = (await Promise.all(calls)).filter(Boolean);
  if (!results.length) throw new Error(\`No valid prices for \${symbol}\`);

  const prices = results.map(r => r.price);
  const median = prices.sort((a,b)=>a-b)[Math.floor(prices.length/2)];

  const sane = results.filter(r => Math.abs(r.price - median)/median < 0.1);

  for (const ex of ["binance","coinbase","kraken"]) {
    const m = sane.find(r => r.name === ex);
    if (m) {
      const out = { symbol, quote, price: m.price, source: ex, ts: Date.now() };
      CACHE.set(key, out);
      return out;
    }
  }

  const out = { symbol, quote, price: median, source:"median", ts: Date.now() };
  CACHE.set(key, out);
  return out;
}
EOF

cat > analytics/aggregator/server.ts << 'EOF'
import express from 'express';
import cors from 'cors';
import { getAggregatedPrice } from './aggregator';

const app = express();
app.use(cors());

app.get('/api/price', async (req, res) => {
  try {
    const symbol = req.query.symbol;
    const quote  = req.query.quote || "USD";
    const data   = await getAggregatedPrice({ symbol, quote });
    res.json(data);
  } catch (err) {
    res.status(500).json({ error: "Price fetch failed", details: err.message });
  }
});

app.listen(3001, () =>
  console.log("STRATA Aggregator running on :3001")
);
EOF

import { useEffect, useState } from 'react';

export function useSnapshotPrices(refreshMs = 15000) {
  const [data, setData] = useState([]);

  async function load() {
    const r = await fetch('/api/prices?symbols=BTC,ETH,SOL&quote=USD');
    const json = await r.json();
    setData(json.prices);
  }

  useEffect(() => {
    load();
    const t = setInterval(load, refreshMs);
    return () => clearInterval(t);
  }, []);

  return data;
}
