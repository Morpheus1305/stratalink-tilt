## Complete Export Handler Example

```typescript
interface RCLSnapshot {
  meta: {
    contract_version: string;
    generated_at: string;
    time_mode: string;
    authoritative_records: {
      poli_snapshot: string;
      dact_window: string;
      lis_manifest: string;
    };
  };
  supervisory_scope: {
    jurisdiction: string;
    asset_class: string;
    market_type: string;
    dsu: string[];
    version: string;
  };
  coverage: {
    instrument: string;
    venue_count: number;
    liquidity_types: string[];
    coverage_pct: number;
    known_venues: number;
    covered_venues: number;
    last_ingest: string;
  };
  truth: {
    status: string;
    evidence_level: string;
    evidence_description: string;
    validity_window: {
      verified_at: string;
      valid_until: string;
    };
    integrity: {
      status: string;
      data_gaps: number;
      ingestion_latency_p95_ms: number;
      within_bounds: boolean;
      normalization: string;
    };
    degradation_reason?: string;
  };
  provenance: Record<string, any>;
  references: Record<string, string>;
  compliance: {
    annex_a: string;
    annex_b: string;
    access_scope: string;
    ui_status: string;
  };
}

const handleExportJSON = () => {
  // Build your snapshot data from current state
  const snapshot: RCLSnapshot = buildSnapshotData();
  
  // Format with indentation
  const formattedJSON = JSON.stringify(snapshot, null, 2);
  
  // Create filename with timestamp
  const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
  const filename = `rcl-snapshot-${timestamp}.json`;
  
  // Download
  downloadFile(formattedJSON, filename, 'application/json');
};

const downloadFile = (content: string, filename: string, mimeType: string) => {
  const blob = new Blob([content], { type: mimeType });
  const url = URL.createObjectURL(blob);
  const link = document.createElement('a');
  link.href = url;
  link.download = filename;
  document.body.appendChild(link);
  link.click();
  document.body.removeChild(link);
  URL.revokeObjectURL(url);
};
```
