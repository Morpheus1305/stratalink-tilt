# Complete Working PDF Export Implementation

## Complete Working Solution

```typescript
import { jsPDF } from 'jspdf';

interface SnapshotData {
  meta: {
    contract_version: string;
    generated_at: string;
    authoritative_records: {
      poli_snapshot: string;
      dact_window: string;
      lis_manifest: string;
    };
  };
  supervisoryScope: {
    jurisdiction: string;
    asset_class: string;
    market_type: string;
    dsu: string[];
    version: string;
  };
  coverage: {
    instrument: string;
    venue_count: number;
    liquidity_types: string[];
    coverage_pct: number;
    known_venues: number;
    covered_venues: number;
    last_ingest: string;
  };
  truth: {
    status: string;
    evidence_level: string;
    evidence_description: string;
    validity_window: {
      verified_at: string;
      valid_until: string;
    };
    integrity: {
      status: string;
      data_gaps: number;
      ingestion_latency_p95_ms: number;
      within_bounds: boolean;
      normalization: string;
    };
    degradation_reason?: string;
  };
  provenance: Record<string, {
    status: string;
    modules: string[];
    normalization: string;
    last_event: string;
    lis_ref: string;
  }>;
  references: {
    snapshot_ref: string;
    poli_ref: string;
    dact_ref: string;
    lis_ref: string;
  };
  compliance: {
    annex_a: string;
    annex_b: string;
    access_scope: string;
    ui_status: string;
  };
}

export const generateRCLPDF = (data: SnapshotData) => {
  const doc = new jsPDF({
    orientation: 'portrait',
    unit: 'mm',
    format: 'a4'
  });
  
  const pageWidth = doc.internal.pageSize.getWidth();
  const pageHeight = doc.internal.pageSize.getHeight();
  const margin = 20;
  const contentWidth = pageWidth - (2 * margin);
  let currentY = margin;
  
  // Helper: Check if we need a new page
  const checkNewPage = (spaceNeeded: number = 10) => {
    if (currentY + spaceNeeded > pageHeight - margin) {
      doc.addPage();
      currentY = margin;
      return true;
    }
    return false;
  };
  
  // Helper: Add section heading
  const addHeading = (text: string, size: number = 14) => {
    checkNewPage(15);
    doc.setFontSize(size);
    doc.setFont('helvetica', 'bold');
    doc.text(text, margin, currentY);
    currentY += size * 0.5 + 5;
  };
  
  // Helper: Add key-value pair
  const addKeyValue = (key: string, value: string, indent: number = 0) => {
    checkNewPage(8);
    doc.setFontSize(10);
    doc.setFont('helvetica', 'bold');
    const keyText = `${key}:`;
    doc.text(keyText, margin + indent, currentY);
    
    doc.setFont('helvetica', 'normal');
    const keyWidth = doc.getTextWidth(keyText);
    
    // Handle long values with text wrapping
    const maxValueWidth = contentWidth - keyWidth - indent - 5;
    const valueLines = doc.splitTextToSize(value, maxValueWidth);
    
    doc.text(valueLines[0], margin + indent + keyWidth + 2, currentY);
    currentY += 6;
    
    // Add remaining lines if wrapped
    for (let i = 1; i < valueLines.length; i++) {
      checkNewPage(6);
      doc.text(valueLines[i], margin + indent + keyWidth + 2, currentY);
      currentY += 6;
    }
  };
  
  // Helper: Add spacer
  const addSpace = (space: number = 5) => {
    currentY += space;
  };
  
  // ===== START DOCUMENT CONTENT =====
  
  // Title
  doc.setFontSize(18);
  doc.setFont('helvetica', 'bold');
  doc.text('ADGM Regulatory Liquidity Snapshot', margin, currentY);
  currentY += 15;
  
  // Horizontal line
  doc.setLineWidth(0.5);
  doc.line(margin, currentY, pageWidth - margin, currentY);
  currentY += 10;
  
  // Document Metadata Section
  addHeading('Document Metadata', 14);
  addKeyValue('Type', 'Regulatory Liquidity Snapshot');
  addKeyValue('Jurisdiction', data.supervisoryScope.jurisdiction);
  addKeyValue('Contract Version', data.meta.contract_version);
  addKeyValue('Instrument', data.coverage.instrument);
  addKeyValue('Snapshot Reference', data.references.snapshot_ref);
  addKeyValue('Generated At', data.meta.generated_at);
  addSpace(8);
  
  // Supervisory Scope Section
  addHeading('Supervisory Scope (RCL-v0.1)', 14);
  addKeyValue('Jurisdiction', data.supervisoryScope.jurisdiction);
  addKeyValue('Asset Class', data.supervisoryScope.asset_class);
  addKeyValue('Market Type', data.supervisoryScope.market_type);
  addKeyValue('Declared Supervisory Universe (DSU)', 
    data.supervisoryScope.dsu.join(', '));
  addKeyValue('Version', data.supervisoryScope.version);
  addSpace(8);
  
  // Coverage Analysis Section
  addHeading('Coverage Analysis', 14);
  addKeyValue('Instrument', data.coverage.instrument);
  addKeyValue('Venue Count', data.coverage.venue_count.toString());
  addKeyValue('Liquidity Types', data.coverage.liquidity_types.join(', '));
  addKeyValue('Coverage Percentage', `${data.coverage.coverage_pct}%`);
  addKeyValue('Known Venues', data.coverage.known_venues.toString());
  addKeyValue('Covered Venues', data.covered_venues.toString());
  addKeyValue('Last Ingest', data.coverage.last_ingest);
  addSpace(8);
  
  // Liquidity Truth Status Section
  addHeading('Liquidity Truth Status', 14);
  addKeyValue('Status', data.truth.status);
  addKeyValue('Evidence Level', 
    `${data.truth.evidence_level} (${data.truth.evidence_description})`);
  addKeyValue('Verified At', data.truth.validity_window.verified_at);
  addKeyValue('Valid Until', data.truth.validity_window.valid_until);
  
  if (data.truth.degradation_reason) {
    addKeyValue('Degradation Reason', data.truth.degradation_reason);
  }
  addSpace(8);
  
  // Integrity Metrics Section
  addHeading('Integrity Metrics', 14);
  addKeyValue('Integrity Status', data.truth.integrity.status);
  addKeyValue('Data Gaps', data.truth.integrity.data_gaps.toString());
  addKeyValue('Ingestion Latency (p95)', 
    `${data.truth.integrity.ingestion_latency_p95_ms} ms`);
  addKeyValue('Within Bounds', data.truth.integrity.within_bounds ? 'Yes' : 'No');
  addKeyValue('Normalization', data.truth.integrity.normalization);
  addSpace(8);
  
  // Venue Provenance Section
  addHeading('Venue Provenance', 14);
  
  Object.entries(data.provenance).forEach(([venue, info]) => {
    checkNewPage(30); // Ensure enough space for venue block
    
    doc.setFontSize(12);
    doc.setFont('helvetica', 'bold');
    doc.text(venue.toUpperCase(), margin, currentY);
    currentY += 8;
    
    addKeyValue('Connection Status', info.status, 5);
    addKeyValue('Modules', info.modules.join(', '), 5);
    addKeyValue('Normalization', info.normalization, 5);
    addKeyValue('Last Event', info.last_event, 5);
    addKeyValue('LIS Reference', info.lis_ref, 5);
    addSpace(5);
  });
  
  addSpace(3);
  
  // Reference Identifiers Section
  addHeading('Reference Identifiers', 14);
  addKeyValue('Snapshot Reference', data.references.snapshot_ref);
  addKeyValue('PoLi Reference', data.references.poli_ref);
  addKeyValue('DACT Reference', data.references.dact_ref);
  addKeyValue('LIS Reference', data.references.lis_ref);
  addSpace(8);
  
  // Authoritative Records Section
  addHeading('Authoritative Records (Official)', 14);
  addKeyValue('PoLi Snapshot', data.meta.authoritative_records.poli_snapshot);
  addKeyValue('DACT Window', data.meta.authoritative_records.dact_window);
  addKeyValue('LIS Manifest', data.meta.authoritative_records.lis_manifest);
  addSpace(8);
  
  // Compliance Section
  addHeading('Regulatory Compliance', 14);
  addKeyValue('Annex A', data.compliance.annex_a);
  addKeyValue('Annex B', data.compliance.annex_b);
  addKeyValue('Access Scope', data.compliance.access_scope);
  addKeyValue('UI Status', data.compliance.ui_status);
  addSpace(10);
  
  // Important Notice Section
  checkNewPage(40);
  doc.setFontSize(12);
  doc.setFont('helvetica', 'bold');
  doc.text('Important Notice', margin, currentY);
  currentY += 8;
  
  doc.setFontSize(10);
  doc.setFont('helvetica', 'normal');
  const noticeText = 'This PDF is a rendered view of the RCL regulatory snapshot. For official records and authoritative evidence, refer to the machine-verifiable artifacts referenced above (PoLi snapshots, DACT artifacts, LIS manifests).';
  const noticeLines = doc.splitTextToSize(noticeText, contentWidth);
  doc.text(noticeLines, margin, currentY);
  currentY += noticeLines.length * 6;
  addSpace(5);
  
  doc.setFont('helvetica', 'bold');
  doc.text('UI Status: Non-authoritative (display only)', margin, currentY);
  currentY += 6;
  doc.setFont('helvetica', 'normal');
  doc.text('Official Records: Machine-verifiable PoLi, DACT, and LIS artifacts', 
    margin, currentY);
  currentY += 10;
  
  doc.setFontSize(9);
  doc.text('This document was generated from RCL-v0.1 in accordance with:', 
    margin, currentY);
  currentY += 5;
  doc.text('• Annex A: Declared Supervisory Universe', margin, currentY);
  currentY += 5;
  doc.text('• Annex B: Venue Expansion Playbook', margin, currentY);
  
  // Generate filename with timestamp
  const timestamp = new Date().toISOString()
    .replace(/[:.]/g, '-')
    .slice(0, -5);
  const filename = `rcl-snapshot-${timestamp}.pdf`;
  
  // Save the PDF
  doc.save(filename);
  
  return filename;
};
```
