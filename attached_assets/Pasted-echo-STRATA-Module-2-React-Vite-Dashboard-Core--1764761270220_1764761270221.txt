# 3. src/App.tsx — core layout + data loading
############################################
cat > src/App.tsx << 'EOF'
import React, { useEffect, useState } from "react";

type PriceRow = {
  symbol: string;
  price: number;
  source: string;
};

type StressState = {
  stress_score: number;
  regime: string;
  drivers: string[];
  depth?: any;
  funding?: any;
  liquidations?: any;
};

type SummaryState = {
  dominant_factor: string;
  market_regime: string;
};

const TOKENS = ["BTC", "ETH", "SOL"];

export default function App() {
  const [prices, setPrices] = useState<PriceRow[]>([]);
  const [stress, setStress] = useState<StressState | null>(null);
  const [summary, setSummary] = useState<SummaryState | null>(null);
  const [loading, setLoading] = useState(true);

  // ---- Load dashboard data ----
  async function loadAll() {
    try {
      const [pRes, sRes, mRes] = await Promise.all([
        fetch("/api/prices?symbols=BTC,ETH,SOL"),
        fetch("/api/stress/full"),
        fetch("/api/summary"),
      ]);

      const pJson = await pRes.json();
      const sJson = await sRes.json();
      const mJson = await mRes.json();

      setPrices(pJson?.prices || []);
      setStress(sJson);
      setSummary(mJson);
    } catch (err) {
      console.error("Dashboard load error:", err);
    } finally {
      setLoading(false);
    }
  }

  // Poll every 15s
  useEffect(() => {
    loadAll();
    const id = setInterval(loadAll, 15000);
    return () => clearInterval(id);
  }, []);

  return (
    <div style={{ padding: "24px 32px", maxWidth: 1200, margin: "0 auto" }}>
      <header style={{ marginBottom: 24 }}>
        <div style={{ fontSize: 12, letterSpacing: 3, color: "#8ea3c7" }}>
          STRATA • ANALYTICS
        </div>
        <h1 style={{ margin: "4px 0 0", fontSize: 26 }}>
          Daily Crypto Market Structure Attribution
        </h1>
      </header>

      {loading && <div>Loading...</div>}

      {/* Placeholder Panels — Filled in Module 3 */}
      <section
        style={{
          background: "#0b1020",
          padding: 16,
          borderRadius: 12,
          marginBottom: 16,
          border: "1px solid #1a2138",
        }}
      >
        <div style={{ color: "#8ea3c7", fontSize: 12 }}>MARKET SNAPSHOT</div>
        <pre style={{ fontSize: 12, marginTop: 12 }}>
          {JSON.stringify(prices, null, 2)}
        </pre>
      </section>

      <section
        style={{
          background: "#0b1020",
          padding: 16,
          borderRadius: 12,
          marginBottom: 16,
          border: "1px solid #1a2138",
        }}
      >
        <div style={{ color: "#8ea3c7", fontSize: 12 }}>STRESS PANEL</div>
        <pre style={{ fontSize: 12, marginTop: 12 }}>
          {JSON.stringify(stress, null, 2)}
        </pre>
      </section>

      <section
        style={{
          background: "#0b1020",
          padding: 16,
          borderRadius: 12,
          marginBottom: 16,
          border: "1px solid #1a2138",
        }}
      >
        <div style={{ color: "#8ea3c7", fontSize: 12 }}>SUMMARY</div>
        <pre style={{ fontSize: 12, marginTop: 12 }}>
          {JSON.stringify(summary, null, 2)}
        </pre>
      </section>

      <footer style={{ marginTop: 24, fontSize: 11, color: "#6d7da2" }}>
        STRATA • Liquidity Truth Layer — Internal analytics prototype.
      </footer>
    </div>
  );
}
EOF
