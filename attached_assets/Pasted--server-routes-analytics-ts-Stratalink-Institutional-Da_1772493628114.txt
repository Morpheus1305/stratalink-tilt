/**
 * server/routes/analytics.ts
 *
 * Stratalink Institutional Dashboard — Analytics API Route
 *
 * Register in your main Express app:
 *   import analyticsRouter from './routes/analytics';
 *   app.use('/api/analytics', analyticsRouter);
 *
 * Endpoints:
 *   GET /api/analytics/snapshot/:symbol  — full TsleAggregate + L5F composite
 *   GET /api/analytics/health            — service liveness check
 */

import { Router, Request, Response } from 'express';
import { computeAnalyticsSnapshot, TsleAggregate } from '../services/analytics-layer';

const router = Router();

// ── Simple in-process cache ───────────────────────────────────────────────────
// Recomputing all five L5F factors on every request is cheap, but the dashboard
// polls at 5-second intervals across multiple connected clients. A 1-second
// cache prevents redundant work without introducing stale data risk.

interface CacheEntry {
  aggregate: TsleAggregate;
  computedAt: number;
}

const cache = new Map<string, CacheEntry>();
const CACHE_TTL_MS = 1_000;

function getCached(symbol: string): TsleAggregate | null {
  const entry = cache.get(symbol);
  if (!entry) return null;
  if (Date.now() - entry.computedAt > CACHE_TTL_MS) {
    cache.delete(symbol);
    return null;
  }
  return entry.aggregate;
}

function setCache(symbol: string, aggregate: TsleAggregate): void {
  cache.set(symbol, { aggregate, computedAt: Date.now() });
}

// Extend this list to cover all symbols your TSLE feeds emit.
const VALID_SYMBOLS = new Set(['BTC', 'ETH', 'SOL']);

// ── GET /api/analytics/snapshot/:symbol ──────────────────────────────────────

router.get('/snapshot/:symbol', (req: Request, res: Response) => {
  const symbol = String(req.params.symbol ?? '').toUpperCase();

  if (!VALID_SYMBOLS.has(symbol)) {
    return res.status(400).json({
      error: `Invalid symbol. Must be one of: ${[...VALID_SYMBOLS].join(', ')}`,
    });
  }

  try {
    // Return cached result if still fresh
    const cached = getCached(symbol);
    if (cached) {
      return res.json({ ok: true, cached: true, aggregate: cached });
    }

    const aggregate = computeAnalyticsSnapshot(symbol);

    if (!aggregate) {
      // Buffer is present but empty — either the feed hasn't started yet
      // or the venue hasn't emitted a snapshot for this symbol.
      return res.status(503).json({
        ok: false,
        error: `No buffer data for ${symbol}. Feed may still be warming up.`,
      });
    }

    setCache(symbol, aggregate);
    return res.json({ ok: true, cached: false, aggregate });

  } catch (err: any) {
    console.error('[Analytics] /snapshot error:', err);
    return res.status(500).json({
      ok: false,
      error: 'Failed to compute analytics snapshot',
    });
  }
});

// ── GET /api/analytics/health ────────────────────────────────────────────────

router.get('/health', (_req: Request, res: Response) => {
  return res.json({
    ok: true,
    service: 'Stratalink Analytics Layer',
    version: '0.1.0',
    timestamp: Date.now(),
    cachedSymbols: [...cache.keys()],
    supportedSymbols: [...VALID_SYMBOLS],
  });
});

export default router;
