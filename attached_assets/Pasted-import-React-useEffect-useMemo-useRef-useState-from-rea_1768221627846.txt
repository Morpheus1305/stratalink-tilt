import React, { useEffect, useMemo, useRef, useState } from "react";
import { useQuery } from "@tanstack/react-query";

type Venue = "binance" | "coinbase" | "kraken";
type EventType = "trade" | "top_of_book" | "depth" | "heartbeat" | "error";

type LTREvent = {
  event_id?: string;
  id?: string;
  ts_event?: number | null;
  ts_ingest?: number | null;
  venue?: Venue | string;
  symbol?: string;
  type?: EventType | string;        // <-- your API uses `type`
  event_type?: EventType | string;  // <-- tolerate both
  seq?: number | string | null;
  payload?: Record<string, any>;
  [k: string]: any;
};

type LatestResponse = {
  server_ts?: number;
  events?: LTREvent[];
  data?: LTREvent[]; // tolerate alternative shapes
};

type TapeHealthVenue = {
  status: "ok" | "degraded" | "down" | "unknown";
  last_ts_ingest: number | null;
  msg_rate_5s?: number | null;
  lag_ms?: number | null;
  errors_5m?: number | null;
};

type TapeHealthResponse = {
  server_ts: number;
  venues: Record<string, TapeHealthVenue>;
};

function fmtTimeMs(ts: number | null | undefined) {
  if (!ts || Number.isNaN(ts)) return "—";
  const d = new Date(ts);
  const hh = String(d.getHours()).padStart(2, "0");
  const mm = String(d.getMinutes()).padStart(2, "0");
  const ss = String(d.getSeconds()).padStart(2, "0");
  const ms = String(d.getMilliseconds()).padStart(3, "0");
  return `${hh}:${mm}:${ss}.${ms}`;
}

function clamp(n: number, min: number, max: number) {
  return Math.max(min, Math.min(max, n));
}

function safeNum(x: any): number | null {
  const n = Number(x);
  return Number.isFinite(n) ? n : null;
}

function computeSpread(payload: any): number | null {
  const bid = safeNum(payload?.bid_px);
  const ask = safeNum(payload?.ask_px);
  if (bid == null || ask == null) return null;
  return ask - bid;
}

function normalizeEvent(e: LTREvent) {
  const event_id = e.event_id ?? e.id ?? `${e.venue ?? "?"}-${e.symbol ?? "?"}-${e.ts_ingest ?? e.ts_event ?? Date.now()}`;
  const ts_ingest = (e.ts_ingest ?? e.ingest_ts ?? e.ts ?? null) as number | null;
  const ts_event = (e.ts_event ?? e.event_ts ?? null) as number | null;
  const venue = (e.venue ?? e.exchange ?? "unknown") as string;
  const symbol = (e.symbol ?? e.pair ?? "—") as string;
  const event_type = (e.event_type ?? e.type ?? "unknown") as string;
  return { ...e, event_id, ts_ingest, ts_event, venue, symbol, event_type };
}

function keyFields(e: ReturnType<typeof normalizeEvent>): string {
  const p = e.payload || {};
  switch (e.event_type) {
    case "top_of_book": {
      const bidPx = safeNum(p.bid_px);
      const bidSz = safeNum(p.bid_sz);
      const askPx = safeNum(p.ask_px);
      const askSz = safeNum(p.ask_sz);
      const spr = computeSpread(p);
      const left = bidPx != null ? `${bidPx}${bidSz != null ? ` × ${bidSz}` : ""}` : "—";
      const right = askPx != null ? `${askPx}${askSz != null ? ` × ${askSz}` : ""}` : "—";
      const spread = spr != null ? ` | spr ${spr}` : "";
      return `${left} | ${right}${spread}`;
    }
    case "trade": {
      const px = safeNum(p.px ?? p.price);
      const sz = safeNum(p.sz ?? p.size ?? p.qty);
      const side = (p.side ?? p.taker_side ?? "").toString().toUpperCase();
      const core = `${px ?? "—"}${sz != null ? ` × ${sz}` : ""}`;
      return side ? `${core} (${side})` : core;
    }
    case "depth": {
      const levels = safeNum(p.levels ?? p.depth_levels ?? p.n_levels);
      const deltaCount = safeNum(p.delta_count ?? p.changes?.length);
      const spr = computeSpread(p);
      const bits = [
        levels != null ? `levels=${levels}` : null,
        deltaCount != null ? `Δ=${deltaCount}` : null,
        spr != null ? `spr=${spr}` : null,
      ].filter(Boolean);
      return bits.length ? bits.join(" | ") : "depth update";
    }
    case "error": {
      const code = p.code ?? p.error_code ?? "ERR";
      const msg = p.message ?? p.error ?? "unknown error";
      return `${code}: ${String(msg).slice(0, 140)}`;
    }
    case "heartbeat":
      return "ok";
    default:
      return "";
  }
}

function buildLatestUrl(params: {
  symbol: string;
  venues: Venue[];
  types: EventType[];
  limit: number;
  windowSec: number;
}) {
  // Your API supports venue/type as singular filters.
  // For multi-select, we’ll call /api/tape/latest once and filter client-side.
  // (Keeps server unchanged; no mocks.)
  const usp = new URLSearchParams();
  if (params.symbol) usp.set("symbol", params.symbol);
  usp.set("limit", String(params.limit));

  // Use time window via since/until if the server supports it
  const until = Date.now();
  const since = until - params.windowSec * 1000;
  usp.set("since", String(since));
  usp.set("until", String(until));

  return `/api/tape/latest?${usp.toString()}`;
}

async function fetchJson<T>(url: string): Promise<T> {
  const r = await fetch(url, { headers: { Accept: "application/json" } });
  if (!r.ok) {
    const text = await r.text().catch(() => "");
    throw new Error(`HTTP ${r.status} ${r.statusText}${text ? ` — ${text}` : ""}`.trim());
  }
  return (await r.json()) as T;
}

export default function TapePage() {
  const [symbol, setSymbol] = useState<string>("BTC-USD");

  const [venues, setVenues] = useState<Record<Venue, boolean>>({
    binance: true,
    coinbase: true,
    kraken: true,
  });

  const [types, setTypes] = useState<Record<EventType, boolean>>({
    top_of_book: true,
    trade: true,
    depth: false,
    heartbeat: false,
    error: true,
  });

  const [windowSec, setWindowSec] = useState<number>(60);
  const [limit, setLimit] = useState<number>(200);

  const [search, setSearch] = useState<string>("");
  const [paused, setPaused] = useState<boolean>(false);
  const [autoScroll, setAutoScroll] = useState<boolean>(true);
  const [advanced, setAdvanced] = useState<boolean>(false);

  const [expanded, setExpanded] = useState<Record<string, boolean>>({});

  const selectedVenues = useMemo(
    () => (Object.keys(venues) as Venue[]).filter((v) => venues[v]),
    [venues]
  );
  const selectedTypes = useMemo(
    () => (Object.keys(types) as EventType[]).filter((t) => types[t]),
    [types]
  );

  const latestUrl = useMemo(
    () =>
      buildLatestUrl({
        symbol: symbol.trim(),
        venues: selectedVenues,
        types: selectedTypes,
        limit,
        windowSec,
      }),
    [symbol, selectedVenues, selectedTypes, limit, windowSec]
  );

  const latestQuery = useQuery({
    queryKey: ["tape-latest", latestUrl],
    queryFn: () => fetchJson<LatestResponse>(latestUrl),
    enabled: !paused && symbol.trim().length > 0,
    refetchInterval: paused ? false : 750,
    refetchIntervalInBackground: true,
    staleTime: 0,
    retry: 2,
  });

  const healthQuery = useQuery({
    queryKey: ["tape-health"],
    queryFn: () => fetchJson<TapeHealthResponse>("/api/tape/health"),
    refetchInterval: 2000,
    refetchIntervalInBackground: true,
    staleTime: 0,
    retry: 2,
  });

  const raw = latestQuery.data?.events ?? latestQuery.data?.data ?? [];
  const events = useMemo(() => {
    const s = search.trim().toLowerCase();

    return raw
      .map(normalizeEvent)
      .filter((e) => selectedVenues.includes((e.venue as Venue) ?? "binance"))
      .filter((e) => selectedTypes.includes((e.event_type as EventType) ?? "trade"))
      .filter((e) => {
        if (!s) return true;
        const hay = `${e.event_id} ${e.venue} ${e.symbol} ${e.event_type}`.toLowerCase();
        return hay.includes(s);
      })
      .sort((a, b) => (Number(b.ts_ingest ?? 0) - Number(a.ts_ingest ?? 0)));
  }, [raw, search, selectedVenues, selectedTypes]);

  // Autoscroll (newest-first)
  const listRef = useRef<HTMLDivElement | null>(null);
  useEffect(() => {
    const el = listRef.current;
    if (!el) return;

    const onScroll = () => {
      const atTop = el.scrollTop <= 10;
      if (!atTop && autoScroll) setAutoScroll(false);
    };

    el.addEventListener("scroll", onScroll, { passive: true });
    return () => el.removeEventListener("scroll", onScroll);
  }, [autoScroll]);

  useEffect(() => {
    if (!autoScroll) return;
    const el = listRef.current;
    if (!el) return;
    el.scrollTop = 0;
  }, [events.length, autoScroll]);

  function toggleVenue(v: Venue) {
    setVenues((prev) => ({ ...prev, [v]: !prev[v] }));
  }

  function toggleType(t: EventType) {
    setTypes((prev) => ({ ...prev, [t]: !prev[t] }));
  }

  function toggleExpand(eventId: string) {
    setExpanded((prev) => ({ ...prev, [eventId]: !prev[eventId] }));
  }

  function jumpToLatest() {
    const el = listRef.current;
    if (!el) return;
    el.scrollTop = 0;
    setAutoScroll(true);
  }

  async function downloadJson() {
    const blob = new Blob(
      [JSON.stringify({ symbol, filters: { venues, types, search, windowSec, limit }, events }, null, 2)],
      { type: "application/json" }
    );
    const a = document.createElement("a");
    a.href = URL.createObjectURL(blob);
    a.download = `tape_${symbol}_${Date.now()}.json`;
    a.click();
    URL.revokeObjectURL(a.href);
  }

  function statusBoxClass(status: string) {
    const s = status.toLowerCase();
    if (s === "down") return "border border-neutral-800 opacity-70";
    if (s === "degraded") return "border border-neutral-700";
    if (s === "ok") return "border border-neutral-700";
    return "border border-neutral-800 opacity-70";
  }

  const hb = healthQuery.data?.venues ?? {};
  const now = Date.now();
  const showEmpty = !latestQuery.isLoading && events.length === 0;

  return (
    <div className="p-4 space-y-4">
      <div className="flex flex-col gap-2 md:flex-row md:items-end md:justify-between">
        <div>
          <h1 className="text-xl font-semibold">Live Tape</h1>
          <div className="text-sm opacity-80">Live events from /api/tape (no mocks)</div>
        </div>

        <div className="flex flex-wrap gap-2 items-center">
          <span className="text-xs px-2 py-1 rounded border border-neutral-800 opacity-80">
            LIVE (Polling)
          </span>

          <button
            onClick={() => setPaused((p) => !p)}
            className="text-sm px-3 py-1.5 rounded border border-neutral-800 hover:opacity-90"
          >
            {paused ? "Resume" : "Pause"}
          </button>

          <button
            onClick={() => setAdvanced((a) => !a)}
            className="text-sm px-3 py-1.5 rounded border border-neutral-800 hover:opacity-90"
          >
            {advanced ? "Advanced: On" : "Advanced: Off"}
          </button>

          <button
            onClick={downloadJson}
            className="text-sm px-3 py-1.5 rounded border border-neutral-800 hover:opacity-90"
          >
            Export JSON
          </button>
        </div>
      </div>

      {/* Heartbeat strip (requires /api/tape/health) */}
      <div className="grid grid-cols-1 md:grid-cols-3 gap-2">
        {(["binance", "coinbase", "kraken"] as Venue[]).map((v) => {
          const h = hb[v];
          const last = h?.last_ts_ingest ?? null;
          const ageSec = last ? Math.max(0, Math.round(((now - last) / 100)) / 10) : null;
          const status = h?.status ?? "unknown";
          return (
            <div key={v} className={`rounded p-3 ${statusBoxClass(status)}`}>
              <div className="flex items-center justify-between">
                <div className="font-medium">{v}</div>
                <div className="text-xs opacity-80">{String(status).toUpperCase()}</div>
              </div>
              <div className="mt-1 text-sm opacity-90">
                Last msg: {ageSec != null ? `${ageSec}s ago` : "—"}
              </div>
              <div className="mt-1 text-xs opacity-80 flex gap-3 flex-wrap">
                <span>Rate: {h?.msg_rate_5s ?? "—"}</span>
                <span>Lag: {h?.lag_ms ?? "—"}ms</span>
                <span>Err(5m): {h?.errors_5m ?? "—"}</span>
              </div>
            </div>
          );
        })}
      </div>

      {/* Filters */}
      <div className="rounded border border-neutral-900 p-3 space-y-3">
        <div className="flex flex-col md:flex-row md:items-center gap-3">
          <div className="flex items-center gap-2">
            <div className="text-sm opacity-80 w-16">Symbol</div>
            <input
              value={symbol}
              onChange={(e) => setSymbol(e.target.value)}
              className="px-2 py-1.5 rounded border border-neutral-800 bg-transparent w-44"
              placeholder="BTC-USD"
            />
          </div>

          <div className="flex items-center gap-2 flex-wrap">
            <div className="text-sm opacity-80 w-16">Venues</div>
            {(["binance", "coinbase", "kraken"] as Venue[]).map((v) => (
              <button
                key={v}
                onClick={() => toggleVenue(v)}
                className={`text-sm px-2.5 py-1.5 rounded border ${
                  venues[v] ? "border-neutral-700" : "border-neutral-900 opacity-60"
                }`}
              >
                {v}
              </button>
            ))}
          </div>

          <div className="flex items-center gap-2 flex-wrap">
            <div className="text-sm opacity-80 w-16">Types</div>
            {(Object.keys(types) as EventType[]).map((t) => (
              <button
                key={t}
                onClick={() => toggleType(t)}
                className={`text-sm px-2.5 py-1.5 rounded border ${
                  types[t] ? "border-neutral-700" : "border-neutral-900 opacity-60"
                }`}
              >
                {t}
              </button>
            ))}
          </div>
        </div>

        <div className="flex flex-col md:flex-row md:items-center gap-3">
          <div className="flex items-center gap-2">
            <div className="text-sm opacity-80 w-16">Window</div>
            <select
              value={windowSec}
              onChange={(e) => setWindowSec(Number(e.target.value))}
              className="px-2 py-1.5 rounded border border-neutral-800 bg-transparent"
            >
              <option value={30}>30s</option>
              <option value={60}>60s</option>
              <option value={300}>5m</option>
            </select>
          </div>

          <div className="flex items-center gap-2">
            <div className="text-sm opacity-80 w-16">Limit</div>
            <select
              value={limit}
              onChange={(e) => setLimit(clamp(Number(e.target.value), 50, 500))}
              className="px-2 py-1.5 rounded border border-neutral-800 bg-transparent"
            >
              <option value={100}>100</option>
              <option value={200}>200</option>
              <option value={300}>300</option>
              <option value={500}>500</option>
            </select>
          </div>

          <div className="flex items-center gap-2 flex-1">
            <div className="text-sm opacity-80 w-16">Search</div>
            <input
              value={search}
              onChange={(e) => setSearch(e.target.value)}
              className="px-2 py-1.5 rounded border border-neutral-800 bg-transparent w-full"
              placeholder="event_id, venue, type…"
            />
          </div>

          <div className="flex items-center gap-2">
            <button
              onClick={() => setAutoScroll((a) => !a)}
              className="text-sm px-3 py-1.5 rounded border border-neutral-800 hover:opacity-90"
            >
              Auto-scroll: {autoScroll ? "On" : "Off"}
            </button>
            <button
              onClick={jumpToLatest}
              className="text-sm px-3 py-1.5 rounded border border-neutral-800 hover:opacity-90"
            >
              Jump to latest
            </button>
          </div>
        </div>

        <div className="text-xs opacity-80 flex flex-wrap gap-3">
          <span>Latest: {latestQuery.isLoading ? "loading…" : latestQuery.isError ? "error" : "ok"}</span>
          <span>Health: {healthQuery.isLoading ? "loading…" : healthQuery.isError ? "error" : "ok"}</span>
          {latestQuery.isError && <span className="opacity-90">({(latestQuery.error as Error)?.message})</span>}
          {healthQuery.isError && <span className="opacity-90">({(healthQuery.error as Error)?.message})</span>}
        </div>
      </div>

      {/* Tape table */}
      <div className="rounded border border-neutral-900 overflow-hidden">
        <div className="px-3 py-2 border-b border-neutral-900 flex items-center justify-between">
          <div className="text-sm opacity-80">
            Showing <span className="opacity-100">{events.length}</span> events (filtered)
          </div>
          <div className="text-xs opacity-70">Ordered by ingest time (newest first)</div>
        </div>

        {showEmpty ? (
          <div className="p-6 text-sm opacity-80">
            No events received for current filters. Check venue status above, or widen types/venues.
          </div>
        ) : (
          <div ref={listRef} className="max-h-[70vh] overflow-auto">
            <table className="w-full text-sm">
              <thead className="sticky top-0 bg-neutral-950">
                <tr className="text-left border-b border-neutral-900">
                  <th className="p-2 w-8"></th>
                  <th className="p-2">Time (ingest)</th>
                  {advanced && <th className="p-2">Time (event)</th>}
                  <th className="p-2">Venue</th>
                  <th className="p-2">Symbol</th>
                  <th className="p-2">Type</th>
                  {advanced && <th className="p-2">Seq</th>}
                  {advanced && <th className="p-2">Latency</th>}
                  <th className="p-2">Key fields</th>
                </tr>
              </thead>

              <tbody>
                {events.map((rawEvent) => {
                  const e = normalizeEvent(rawEvent);
                  const isOpen = !!expanded[e.event_id];
                  const latency =
                    e.ts_event && Number.isFinite(e.ts_event) && e.ts_ingest
                      ? Number(e.ts_ingest) - Number(e.ts_event)
                      : null;

                  return (
                    <React.Fragment key={e.event_id}>
                      <tr className="border-b border-neutral-950 align-top">
                        <td className="p-2">
                          <button
                            onClick={() => toggleExpand(e.event_id)}
                            className="px-2 py-0.5 rounded border border-neutral-900 hover:opacity-90 text-xs"
                            aria-label="Toggle row"
                          >
                            {isOpen ? "–" : "+"}
                          </button>
                        </td>

                        <td className="p-2 font-mono text-xs">{fmtTimeMs(e.ts_ingest ?? null)}</td>
                        {advanced && <td className="p-2 font-mono text-xs">{fmtTimeMs(e.ts_event ?? null)}</td>}

                        <td className="p-2">{String(e.venue)}</td>
                        <td className="p-2">{e.symbol}</td>
                        <td className="p-2">{String(e.event_type)}</td>

                        {advanced && <td className="p-2 font-mono text-xs">{e.seq ?? "—"}</td>}
                        {advanced && (
                          <td className="p-2 font-mono text-xs">
                            {latency != null && Number.isFinite(latency) ? `${latency}ms` : "—"}
                          </td>
                        )}

                        <td className="p-2 font-mono text-xs whitespace-pre-wrap break-words">
                          {keyFields(e)}
                        </td>
                      </tr>

                      {isOpen && (
                        <tr className="border-b border-neutral-950">
                          <td className="p-2" />
                          <td className="p-2" colSpan={advanced ? 8 : 6}>
                            <div className="rounded border border-neutral-900 p-3">
                              <div className="text-xs opacity-80 mb-2">
                                event_id: <span className="opacity-100 font-mono">{e.event_id}</span>
                              </div>

                              <pre className="text-xs overflow-auto max-h-[260px] whitespace-pre-wrap break-words">
                                {JSON.stringify(e, null, 2)}
                              </pre>

                              {advanced && (
                                <div className="mt-3 text-xs opacity-80 flex flex-wrap gap-3">
                                  <a
                                    className="underline opacity-80 hover:opacity-100"
                                    href={`/lis?venue=${encodeURIComponent(String(e.venue))}&symbol=${encodeURIComponent(e.symbol)}`}
                                  >
                                    Open in LIS debug
                                  </a>
                                  <a
                                    className="underline opacity-80 hover:opacity-100"
                                    href={`/clt/evidence?event_id=${encodeURIComponent(String(e.event_id))}`}
                                  >
                                    Open evidence
                                  </a>
                                </div>
                              )}
                            </div>
                          </td>
                        </tr>
                      )}
                    </React.Fragment>
                  );
                })}
              </tbody>
            </table>
          </div>
        )}

        <div className="px-3 py-2 border-t border-neutral-900 flex items-center justify-between">
          <div className="text-xs opacity-70">LIVE DATA ONLY • /api/tape/latest</div>
          <div className="text-xs opacity-70">Enable “Advanced” for event time / seq / latency</div>
        </div>
      </div>
    </div>
  );
}
