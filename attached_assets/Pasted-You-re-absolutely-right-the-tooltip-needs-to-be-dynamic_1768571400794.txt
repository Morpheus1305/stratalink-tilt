### React Component
```tsx
interface EvidenceLevelTooltipProps {
  level: EvidenceLevel;
}

const EvidenceLevelTooltip: React.FC<EvidenceLevelTooltipProps> = ({ level }) => {
  const info = EVIDENCE_LEVELS[level];
  
  const getIndicatorIcon = (status: 'complete' | 'partial' | 'warning') => {
    switch (status) {
      case 'complete': return '✓';
      case 'partial': return '◐';
      case 'warning': return '⚠';
    }
  };
  
  const getIndicatorColor = (status: 'complete' | 'partial' | 'warning') => {
    switch (status) {
      case 'complete': return 'text-green-400';
      case 'partial': return 'text-yellow-400';
      case 'warning': return 'text-orange-400';
    }
  };
  
  const getStatusColor = (color: string) => {
    const colors = {
      red: 'text-red-400',
      yellow: 'text-yellow-400',
      green: 'text-green-400',
      blue: 'text-blue-400',
      purple: 'text-purple-400'
    };
    return colors[color as keyof typeof colors];
  };
  
  return (
    <div className="evidence-tooltip max-w-sm p-4 bg-gray-800 rounded-lg shadow-xl">
      <div className="tooltip-header mb-3">
        <div className="font-semibold text-sm text-gray-200 mb-1">
          Evidence Ladder — {info.title}
        </div>
      </div>
      
      <div className="tooltip-indicators space-y-2 mb-3">
        {info.indicators.map((indicator, idx) => (
          <div 
            key={idx}
            className={`flex items-start gap-2 text-xs ${getIndicatorColor(indicator.status)}`}
          >
            <span className="flex-shrink-0">{getIndicatorIcon(indicator.status)}</span>
            <span>{indicator.text}</span>
          </div>
        ))}
      </div>
      
      <div className="tooltip-description text-xs text-gray-400 mb-3 leading-relaxed">
        {info.description}
      </div>
      
      <div className="tooltip-status pt-2 border-t border-gray-700">
        <div className="text-xs">
          <span className="text-gray-500">Status: </span>
          <span className={`font-medium ${getStatusColor(info.statusColor)}`}>
            {info.statusLabel}
          </span>
        </div>
      </div>
    </div>
  );
};
```

## Critical Implementation Note

When PoLi status degrades from L3 → L2 (as shown in your screenshot):

1. **The "DEGRADED" status appears** (yellow/orange badge)
2. **The evidence level changes** to L2
3. **The tooltip MUST update** to show L2 content
4. **Coverage flags appear** explaining WHY it degraded ("elevated_latency")

This creates a coherent narrative:
- Status: DEGRADED
- Evidence Level: L2 (down from L3)
- Reason: "One or more declared supervisory venues reporting with elevated latency"
- Tooltip: Shows L2 indicators with partial sufficiency warnings
